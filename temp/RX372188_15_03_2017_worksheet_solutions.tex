\begin{align*}
1.\hspace{30pt}
 &&  && \displaystyle\frac{x}{2}+29&=31 &&  && \\[15pt]
 &&  && \displaystyle\frac{x}{2}&=31-29 &&  && \\[5pt]
 &&  && \displaystyle\frac{x}{2}&=2 &&  && \\[5pt]
 &&  && x&=2\times2 &&  && \\[5pt]
 &&  && x&=4 &&  && 
\end{align*}
