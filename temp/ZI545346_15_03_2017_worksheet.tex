\begin{minipage}[t]{1.0000\textwidth}
\begin{align*}
1.\hspace{30pt}
 &&  && \left(\displaystyle\frac{18}{x}+3\right)&=-12 &&  && \\[15pt]
 &&  && \displaystyle\frac{18}{x}&=-12-3 &&  && \\[5pt]
 &&  && \displaystyle\frac{18}{x}&=-15 &&  && \\[5pt]
 &&  && x&=\displaystyle\frac{18}{-15} &&  && \\[5pt]
 &&  && x&=-2 &&  && 
\end{align*}
$ x&=-2 $\end{minipage}

2.\hspace{30pt}
 &&  && 66+\displaystyle\frac{8}{x}&=68 &&  && \\[15pt]
 &&  && \displaystyle\frac{8}{x}&=68-66 &&  && \\[5pt]
 &&  && \displaystyle\frac{8}{x}&=2 &&  && \\[5pt]
 &&  && x&=\displaystyle\frac{8}{2} &&  && \\[5pt]
 &&  && x&=4 &&  && \\[5pt]
 &&  && x&=\displaystyle\frac{4}{1} &&  && \\[5pt]
 &&  && x&=4 &&  && 
\end{align*}
$ x&=4 $\end{minipage}
