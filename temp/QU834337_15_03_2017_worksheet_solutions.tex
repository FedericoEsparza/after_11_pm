\begin{align*}
1.\hspace{30pt}
 &&  && 67+\displaystyle\frac{x}{2}&=69 &&  && \\[15pt]
 &&  && \displaystyle\frac{x}{2}&=69-67 &&  && \\[5pt]
 &&  && \displaystyle\frac{x}{2}&=2 &&  && \\[5pt]
 &&  && x&=2\times2 &&  && \\[5pt]
 &&  && x&=4 &&  && \\[5pt]
 &&  && x&=\displaystyle\frac{4}{1} &&  && \\[5pt]
 &&  && x&=4 &&  && 
\end{align*}
$ x&=4 $\end{minipage}

2.\hspace{30pt}
 &&  && \left(6\left(1x\right)+37\right)&=-91 &&  && \\[15pt]
 &&  && 6\left(1x\right)&=-91-37 &&  && \\[5pt]
 &&  && 6\left(1x\right)&=-128 &&  && \\[5pt]
 &&  && x&=\displaystyle\frac{-128}{6} &&  && \\[5pt]
 &&  && x&=-22 &&  && \\[5pt]
 &&  && x&=\displaystyle\frac{-22}{1} &&  && \\[5pt]
 &&  && x&=-22 &&  && 
\end{align*}
$ x&=-22 $\end{minipage}
