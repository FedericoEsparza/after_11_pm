\begin{minipage}[t]{1.0000\textwidth}
\begin{align*}
1.\hspace{30pt}
 &&  && 3\left(1x+89\right)&=285 &&  && \\[15pt]
 &&  && x+89&=\displaystyle\frac{285}{3} &&  && \\[5pt]
 &&  && x+89&=95 &&  && \\[5pt]
 &&  && x&=95-89 &&  && \\[5pt]
 &&  && x&=6 &&  && \\[5pt]
 &&  && x&=\displaystyle\frac{6}{1} &&  && \\[5pt]
 &&  && x&=6 &&  && 
\end{align*}
$ x&=6 $\end{minipage}

\begin{align*}
2.\hspace{30pt}
 &&  && \left(146+9\left(1x\right)\right)&=-209 &&  && \\[15pt]
 &&  && 9\left(1x\right)&=-209-146 &&  && \\[5pt]
 &&  && 9\left(1x\right)&=-355 &&  && \\[5pt]
 &&  && x&=\displaystyle\frac{-355}{9} &&  && \\[5pt]
 &&  && x&=-40 &&  && \\[5pt]
 &&  && x&=\displaystyle\frac{-40}{1} &&  && \\[5pt]
 &&  && x&=-40 &&  && 
\end{align*}
$ x&=-40 $\end{minipage}
